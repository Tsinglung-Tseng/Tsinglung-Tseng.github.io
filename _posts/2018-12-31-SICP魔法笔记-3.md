**Iterative SUM**

(define (sum term a next)
    (define (iter j ans)
        (if (> j b)
            ans
        (iter (next j)
              (* (term j) ans))))
    (iter a 0))


## Finding fix point

y ---- 𝒇 ----> 𝒇(x)

(define (fixed-point f start)
    (define tolerance 0.00001)
    (define (close-enough? u v)
        (< (abs (- u v)) tolerance))
    (define (iter old new)
        (if (close-enough? old new)
            new
            (iter new (f new))))
    (iter start (f start)))
